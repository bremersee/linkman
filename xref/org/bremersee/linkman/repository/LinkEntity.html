39  /**
40   * The link entity.
41   *
42   * @author Christian Bremer
43   */
44  @Document(collection = "links")
45  @TypeAlias("link")
46  @Getter
47  @Setter
48  @ToString
49  @NoArgsConstructor
50  @Validated
51  public class LinkEntity implements Comparable<LinkEntity> {
52  
53    @Id
54    private String id;
55  
56    @Indexed
57    private Set<String> categoryIds = new LinkedHashSet<>();
58  
59    private int order;
60  
61    @NotBlank
62    private String href;
63  
64    private Boolean blank = Boolean.FALSE;
65  
66    @NotBlank
67    @Size(min = 3, max = 75)
68    private String text;
69  
70    private Set<Translation> textTranslations = new LinkedHashSet<>();
71  
72    @NotBlank
73    @Size(max = 255)
74    private String description;
75  
76    private Set<Translation> descriptionTranslations = new LinkedHashSet<>();
77  
78    /**
79     * Gets text.
80     *
81     * @param language the language
82     * @return the text
83     */
84    public String getText(Locale language) {
85      final TwoLetterLanguageCode code = TwoLetterLanguageCode
86          .fromLocale(language, TwoLetterLanguageCode.EN);
87      return Optional.ofNullable(textTranslations)
88          .flatMap(set -> set.stream().filter(entry -> code == entry.getLanguage()).findAny())
89          .map(Translation::getValue)
90          .orElse(text);
91    }
92  
93    /**
94     * Gets description.
95     *
96     * @param language the language
97     * @return the description
98     */
99    public String getDescription(Locale language) {
100     final TwoLetterLanguageCode code = TwoLetterLanguageCode
101         .fromLocale(language, TwoLetterLanguageCode.EN);
102     return Optional.ofNullable(descriptionTranslations)
103         .flatMap(set -> set.stream().filter(entry -> code == entry.getLanguage()).findAny())
104         .map(Translation::getValue)
105         .orElse(description);
106   }

139   @Override
140   public int compareTo(final LinkEntity o) {
141     return compareTo(o, null);
142   }
143 
144   /**
145    * Compare to.
146    *
147    * @param o the other link
148    * @param language the language
149    * @return the result
150    */
151   public int compareTo(final LinkEntity o, final Locale language) {
152     int result = Integer.compare(order, o.order);
153     return result != 0
154         ? result
155         : String.valueOf(getText(language))
156             .compareToIgnoreCase(String.valueOf(o.getText(language)));
157   }
